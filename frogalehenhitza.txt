    I:= 1
    AKAR := KAR
    bitartean (AKAR <> ' ') eta (AKAR <> '.') egin
          LEHENA(I) := AKAR
          I:= I + 1
          irakur (AKAR)
    ambitartean {I aldagaiak gordetako karaktereen kopurua gehi bat dauka}
    LUZERA := I - 1
